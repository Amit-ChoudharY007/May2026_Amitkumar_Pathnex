  ## Terraform - Ec2 with key pair (c6i.8xlarge)

  provider "aws" {
    region = "us-east-1"
  }       

  resource "aws_key_pair" "pathnexkey" {
    key_name    ="pathnexkey"
    public_key  ="ssh-rsa AAAA...."
  }

  resource "aws_instance" "pathnexec2" {
    ami            = "ami-0dcjhdjkhds"
    instance_type  = "c6i.8xlarge"
    key_name       = aws_key_pair.pathnex.key_name 
  }

  tags = {
    Name = "pathnexec2"
  }

  ## Jenkins Pipeline - Parallel Build & test


  pipeline {
    agent any
    stages {
        stage('Parallel Tasks') {
            parallel {
                stage('Build') {
                    steps {
                        sh 'mvn clean compile'
                    }
                }
                stage('Test') {
                    steps {
                        sh 'mvn test'
                    }
                }
            }
        }
    }
  }

    stage: build-test

        - mvn clean compile

    stage: build-test
    image: maven:3.8.1-jdk-17
    script:
      - cd sample-java-app
